void inputArray( int a[], int size){
    int i;
    for( i = 0 ; i < size ; i++ ){
        scanf( "%d", &a[i] ) ;
    }
}
void outputArray( int a[], int size){
    int i;
    for( i = 0; i < size ; i ++ ){
        printf("%d ", a[i] ); 
    }
    printf("\n");
}
int insertArray( int a[], int num, int index, int size){
	int *temp;
    if( index == size ){
        a[index] = num;
        size ++;
    } 
	  else{
			for(int i = size; i > index; i--){
					a[i] = a[i-1];
			}
			a[index] = num;
			size ++;
		}
  return size;
}
int deleteArray( int a[], int index, int size){
  if(index == size-1){
        a[index] = 0;
        size --;
    }
	else{ 
	for(int i = index; i < size - 1; i++){
					a[i] = a[i+1];
			}
			size --;
	}
  return size;
}
