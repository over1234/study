#include <SPI.h>
#include <Wire.h>
#include <Adafruit_GFX.h>
#include <Adafruit_SSD1306.h>
#include <stdint.h>
#include <Adafruit_MLX90614.h>
#define OLED_RESET 4
#define SCREEN_WIDTH 128
#define SCREEN_HEIGHT 64

Adafruit_SSD1306 display(OLED_RESET);
Adafruit_MLX90614 mlx = Adafruit_MLX90614();

#if (SSD1306_LCDHEIGHT != 32)
//#error("Height incorrect, please fix Adafruit_SSD1306.h!");
#endif

void setup() {
  Serial.begin(9600);
  pinMode(8, OUTPUT);
  pinMode(9, OUTPUT);
  pinMode(12, OUTPUT);
  mlx.begin();
  display.begin(SSD1306_SWITCHCAPVCC, 0x3C);
  

}

void loop() {
 display.clearDisplay();
 display.setTextSize(2);
 display.setTextColor(WHITE);
 digitalWrite(8, HIGH);
 delayMicroseconds(10);
 digitalWrite(8, LOW);
 float dist;
 dist = pulseIn(9, HIGH) / 29.0 / 2.0;
 Serial.print("거리 = ");
 Serial.println(dist);
 if(dist < 15)
 {
  digitalWrite(12, HIGH);
  delay(200);
  digitalWrite(12, LOW);
  display.setCursor(8, 10);
  display.print("- START -");
  display.display();
  display.clearDisplay();
  gettemp();
 }
 else
 {
  digitalWrite(12, LOW);
  delay(3000);
 }
}
 void gettemp()
 {
  float temp[4] = {0};
  float ave = 0, sum = 0;
  for(int i = 0; i < 4; i++)
  {
    temp[i] = (mlx.readObjectTempC() + 3);
    sum = sum + temp[i];
    Serial.print("측정중 : ");
    Serial.print(temp[i]);
    delay(800);
  }
  ave = (sum / 4);
  Serial.print("평균온도 = ");
  Serial.println(ave);
  Display_u(ave);
 }
  void Display_u(float tempT)
  {
    if(tempT > 37.5)
    {
      display.setCursor(0, 1);
      display.setTextSize(2);
      display.print("Temp:");
      display.println(tempT);
      display.setCursor(8, 18);
      display.setTextSize(2);
      display.print("DANGEROUS!");
      display.display();
      delay(1000);
      display.clearDisplay();
    }
    else
    {
      display.setCursor(0, 1);
      display.setTextSize(2);
      display.print("Temp:");
      display.println(tempT);
      display.setCursor(12,18);
      display.setTextSize(2);
      display.print("- PASS -");
      display.display();
      delay(1000);
      display.clearDisplay();
    }
  }
